--------------------------------------------------------

-t test.1    -> image name

--------------------------------------------------------

Command:

Explanation:
docker run      -> runs image
--name cont1    -> container name
test.1          -> image name

--------------------------------------------------------

--------------------------------------------------------

--------------------------------------------------------

Command:

Example:

--------------------------------------------------------

Open:

========================================================
STEP 15 - ADD PIPELINE SCRIPT
========================================================

Paste this Jenkins Pipeline Script:

--------------------------------------------------------

pipeline
{

    agent any

    environment
    {
        dockerImage = "ayush05/test.2"
        registry = "ayush05/test.2"
        registryCredential = "jenkin_docker_token"
    }

    stages
    {

        stage('Checkout')
        {

            steps
            {

                checkout scmGit(
                    branches: [[name: '*/main']],
                    extensions: [],
                    userRemoteConfigs: [[
                        url: 'https://github.com/AyushThanthri/dockerprog.git'
                    ]]
                )

            }

        }

        stage('Build Docker Image')
        {

            steps
            {

                script
                {
                    dockerImage = docker.build(registry)
                }

            }

        }

    }

}
